scene.draw = function() {
    scene.update();

    var x = game.mouse.x();
    var y = game.mouse.y();

    layerA.clear();

    layerA.ctx.fillText("Cursor x: " + x,10,20);
    layerA.ctx.fillText("Cursor y: " + y,10,40);

    layerA.ctx.beginPath();
    layerA.ctx.arc(x, y, 10, 0, 2 * Math.PI, "#fff");
    layerA.ctx.fillStyle = "#ccc";
    layerA.ctx.fill();

    scene.requestAnim();
};
