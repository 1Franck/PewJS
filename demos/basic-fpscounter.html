FPS counter



var game = Pew.createProject({
    title : "FPS counter",
    fps: 60,
});

//...

game.createScene('mainScene', (function(){

  //...

  scene.start = function() {
      game.fpsmeter.start();
      //...
  };


  scene.update = function() {
      game.fpsmeter.tick();
      //...
  };

  scene.draw = function() {
      scene.update();

      var fps = game.fpsmeter.current;

      layerA.clear();
      layerA.ctx.fillText('FPS: ' + fps, 10, 20);

      //...

      scene.requestAnim();
  };

//...
})());
